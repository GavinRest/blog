import turtle as t

def koch_curve(length, depth):
    if depth == 0:
        t.forward(length)
    else:
        for angle in [60, -120, 60, 0]:
            koch_curve(length / 3, depth - 1)
            t.left(angle)

t.speed(0)
t.penup()
t.goto(-200, 100)
t.pendown()
t.pensize(2)
for _ in range(3):
    koch_curve(400, 4)  # 递归深度控制复杂度
    t.right(120)
t.done()
